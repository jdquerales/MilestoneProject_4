9def _build_locale_table(filename_or_file): 

10 """ 

11 Parses the FacebookLocales.xml file and builds a dict relating every 

12 available language ('en, 'es, 'zh', ...) with a list of available regions 

13 for that language ('en' -> 'US', 'EN') and an (arbitrary) default region. 

14 """ 

15 # Require the XML parser module only if we want the default mapping 

16 from xml.dom.minidom import parse 

17 

18 dom = parse(filename_or_file) 

19 

20 reps = dom.getElementsByTagName("representation") 

21 locs = map(lambda r: r.childNodes[0].data, reps) 

22 

23 locale_map = {} 

24 for loc in locs: 

25 lang, _, reg = loc.partition("_") 

26 lang_map = locale_map.setdefault(lang, {"regs": [], "default": reg}) 

27 lang_map["regs"].append(reg) 

28 

29 # Default region overrides (arbitrary) 

30 locale_map["en"]["default"] = "US" 

31 # Special case: Use es_ES for Spain and es_LA for everything else 

32 locale_map["es"]["default"] = "LA" 

33 locale_map["zh"]["default"] = "CN" 

34 locale_map["fr"]["default"] = "FR" 

35 locale_map["pt"]["default"] = "PT" 

36 

37 return locale_map

49 def default_locale(request): 

50 """ 

51 Guess an appropriate FB locale based on the active Django locale. 

52 If the active locale is available, it is returned. Otherwise, 

53 it tries to return another locale with the same language. If there 

54 isn't one avaible, 'en_US' is returned. 

55 """ 

56 chosen = "en_US" 

57 language = get_language() 

58 if language: 

59 locale = to_locale(language) 

60 lang, _, reg = locale.partition("_") 

61 

62 lang_map = fb_locales.get(lang) 

63 if lang_map is not None: 

64 if reg in lang_map["regs"]: 

65 chosen = lang + "_" + reg 

66 else: 

67 chosen = lang + "_" + lang_map["default"] 

68 return chosen 

69 

70 return default_locale
